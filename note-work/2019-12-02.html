迁移管理端项目,接口已经有了,新加入导出 excel 功能,尝试使用前端来导出:

let content = `"id","number","string","smile"
"11","111","ssd","嘿嘿"
"22","222","kk","哈哈"
"33","333","sf","呵呵"`;
function loadFile(){
    let aLink = document.createElement('a');
    let blob = new Blob([content], { // 这里传入数组
        type: 'text/plain'
    });
    aLink.download = 'test.csv';
    aLink.href = URL.createObjectURL(blob);
    aLink.click();
    URL.revokeObjectURL(blob);
}
