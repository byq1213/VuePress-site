# 笨办法

循环嵌套，遍历每一项，然后跟当前项的每一项做对比，相同就删除那一项， 有可能造成数组塌陷。

let arr = [1, 2, 3, 4, 2, 1, 3, 4, 9, 3, 3, 4,]
/**
 * 遍历每一项,然后跟当前项的每一项做对比
 * 
 */
function removeDuplication(arr) {
    // 最后一项没有可以对比的
    for (let i = 0; i < arr.length - 1; i++) {
        const item = arr[i];
        for (let x = i + 1; x < arr.length; x++) {
            const xItem = arr[x];

            if (item == xItem) {
                //    相等删除
                arr.splice(x, 1)
                // [ 1, 2, 3, 4, 9, 3 ]
                // 并没有有效删除
                // 直接删除数组造成数组塌陷
                // 删除一项,后面的索引向前移动一位,x++, 所以跳过一项
                x--;
            }
            console.log(arr)
        }
    }
    console.log(arr)
    return arr;
}

// 此方法使用嵌套循环,就执行了30次
removeDuplication(arr)

# 推荐方法

预存到临时对象中，减少循环，遇到重复的将最后的值替换，并删除最后的的索引值，效率很高，但是会打乱数组原先的顺序。

let arr = [1,2,3,4,2,1,3,4,9,3,3,4,]
// ES5 修改原型
Array.prototype.removeDuplication = function(){
    // 声明对象存储数组值
    let obj = {};
    for (let i = 0; i < this.length; i++) {
        const item = this[i];
        if(typeof obj[item] !== 'undefined') {
            // 最后的值覆盖当前值
            this[i] = this[this.length - 1] 
            // 删除最后的值
            this.length--     
            // 下一次循环还是i现在的值       
            i--;
            continue;
        }
        obj[item] = item
    }
}
arr.removeDuplication();
console.log(arr) // => [ 1, 2, 3, 4, 9 ]
